    action: "act",
    request: {

      kind: "evaluate"
    }
  })

**Implementation:**
```javascript
// Client-side text input handler
function sendInput() {
    const text = input.value;
    if (!text) return;
    
    for (const char of text) {
        ws.send(JSON.stringify({ type: 'key', key: char }));
    }
    ws.send(JSON.stringify({ type: 'key', key: 'Enter' }));
    
    status.textContent = 'Sent: "' + text + '"';
    input.value = '';
    setTimeout(() => status.textContent = '', 2000);
}
```
