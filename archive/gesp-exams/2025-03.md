#include <cstdio>
#include <map>

using namespace std;

const int N = 2e5 + 5;

int n;

char s[N];
map <int, int> m;
long long ans;

int main() {
    scanf("%d", &n);
    scanf("%s", s + 1);
    int v = 0;
    m[v]++;
    for (int i = 1; i <= n; i++) {
        v ^= 1 << (s[i] - 'a');

        ans += m[v];
        m[v]++;
    }
    printf("%lld\n", ans);
    return 0;
}
