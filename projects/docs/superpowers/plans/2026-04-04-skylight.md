# Skylight Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Minecraft 1.21.1 Fabric mod with a single toggle keybind that saves original settings, applies peaceful/auto-jump/hidden-hotbar, and restores on toggle-off or disconnect.

**Architecture:** Client-side mod with a state manager, a keybind handler, direct integrated-server access for single-player difficulty changes, a chat-command fallback for multiplayer, an AutoConfig-backed config GUI, and a mixin to cancel hotbar rendering.

**Tech Stack:** Fabric API 0.102.0+, Yarn 1.21.1+build.3, Java 21, AutoConfig 3.4.2, ClothConfig 15.0.127.

import net.minecraft.world.Difficulty;

public class SkylightState {

package com.skylight.mixin;

import com.skylight.state.SkylightState;
import net.minecraft.client.gui.DrawContext;
import net.minecraft.client.gui.hud.InGameHud;
import net.minecraft.client.render.RenderTickCounter;
import org.spongepowered.asm.mixin.Mixin;
import org.spongepowered.asm.mixin.injection.At;
import org.spongepowered.asm.mixin.injection.Inject;
import org.spongepowered.asm.mixin.injection.callback.CallbackInfo;

@Mixin(InGameHud.class)
public class InGameHudMixin {

    @Inject(method = "renderHotbar", at = @At("HEAD"), cancellable = true)
    private void onRenderHotbar(DrawContext context, RenderTickCounter tickCounter, CallbackInfo ci) {
        if (SkylightState.getInstance().isActive()) {
            ci.cancel();
        }
    }
}

package com.skylight;

import com.skylight.config.SkylightConfig;
import com.skylight.state.SkylightState;
import me.shedaniel.autoconfig.AutoConfig;
import me.shedaniel.autoconfig.serializer.GsonConfigSerializer;
import net.fabricmc.api.ClientModInitializer;
import net.fabricmc.fabric.api.client.event.lifecycle.v1.ClientTickEvents;
import net.fabricmc.fabric.api.client.keybinding.v1.KeyBindingHelper;
import net.minecraft.client.MinecraftClient;
import net.minecraft.client.network.ClientPlayNetworkHandler;
import net.minecraft.client.option.GameOptions;
import net.minecraft.client.option.KeyBinding;
import net.minecraft.client.util.InputUtil;
import net.minecraft.server.MinecraftServer;
import net.minecraft.server.integrated.IntegratedServer;
import net.minecraft.text.Text;
import net.minecraft.world.Difficulty;
import org.lwjgl.glfw.GLFW;

public class SkylightMod implements ClientModInitializer {

    private static KeyBinding toggleKey;
    private static SkylightConfig config;

    @Override
    public void onInitializeClient() {
        AutoConfig.register(SkylightConfig.class, GsonConfigSerializer::new);
        config = AutoConfig.getConfigHolder(SkylightConfig.class).getConfig();

        toggleKey = KeyBindingHelper.registerKeyBinding(new KeyBinding(
                "key.skylight.toggle",
                InputUtil.Type.KEYSYM,
                GLFW.GLFW_KEY_K,
                "category.skylight"
        ));

        ClientTickEvents.END_CLIENT_TICK.register(client -> {
            while (toggleKey.wasPressed()) {
                toggle(client);
            }
        });

        ClientPlayConnectionEvents.DISCONNECT.register((handler, client) -> {
            if (SkylightState.getInstance().isActive()) {
                turnOff(client);
            }
        });
    }

    private void toggle(MinecraftClient client) {
        if (client.player == null || client.world == null) return;

        if (SkylightState.getInstance().isActive()) {
            turnOff(client);
            client.player.sendMessage(Text.literal("Skylight Mode: OFF"), true);
        } else {
            turnOn(client);
            client.player.sendMessage(Text.literal("Skylight Mode: ON"), true);
        }
    }

    private void turnOn(MinecraftClient client) {
        SkylightState state = SkylightState.getInstance();
        GameOptions options = client.options;

        state.setOriginalAutoJump(options.autoJump);
        state.setActive(true);

        if (config.autoJumpEnabled) {
            options.autoJump = true;
            options.write();
        }

        if (config.autoPeacefulEnabled && client.world != null) {
            Difficulty current = client.world.getDifficulty();
            state.setOriginalDifficulty(current);
            setDifficulty(client, Difficulty.PEACEFUL);
        }
    }

    private void turnOff(MinecraftClient client) {
        SkylightState state = SkylightState.getInstance();
        GameOptions options = client.options;

        if (config.autoJumpEnabled) {
            options.autoJump = state.getOriginalAutoJump();
            options.write();
        }

        if (config.autoPeacefulEnabled && state.getOriginalDifficulty() != null) {
            setDifficulty(client, state.getOriginalDifficulty());
        }

        state.reset();
    }

    private void setDifficulty(MinecraftClient client, Difficulty difficulty) {
        IntegratedServer integratedServer = client.getServer();
        if (integratedServer != null) {
            for (net.minecraft.server.world.ServerWorld serverWorld : integratedServer.getWorlds()) {
                serverWorld.setDifficulty(difficulty, true);
            }
        } else {
            ClientPlayNetworkHandler handler = client.getNetworkHandler();
            if (handler != null) {
                handler.sendChatCommand("difficulty " + difficulty.getName());
            }
        }
    }
}
